{{/* ========================================================================== comments.html -- Comment system dispatcher Decides which comment system to render (Disqus or Giscus) based on: 1. Per-page front matter: commentsApp="disqus" or default to giscus 2. Per-page comments=true/false to enable/disable entirely 3. params.disableCommentsTypes to disable by content type 4. Disqus requires a shortname in services.disqus config Uses Hugo's Scratch to track state across conditional blocks. ========================================================================== */}} {{- $.Scratch.Set "isComments" true -}} {{- $.Scratch.Set "isDisqus" true -}} {{- $.Scratch.Set "isGiscus" true -}} {{ if eq .Params.commentsApp "disqus" }} {{- $.Scratch.Set "isDisqus" true -}} {{- $.Scratch.Set "isGiscus" false -}} {{ else }} {{- $.Scratch.Set "isDisqus" false -}} {{- $.Scratch.Set "isGiscus" true -}} {{ end }} {{- if and (isset .Params "type") (in .Site.Params.disableCommentsTypes .Params.type) -}} {{- $.Scratch.Set "isComments" false -}} {{- end -}} {{- if and (isset .Params "comments") (eq .Params.comments false) -}} {{- $.Scratch.Set "isComments" false -}} {{- else if and (isset .Params "comments") (eq .Params.comments true) -}} {{- $.Scratch.Set "isComments" true -}} {{- end -}} {{ if not (.Site.Config.Services.Disqus.Shortname) }} {{- $.Scratch.Set "isDisqus" false -}} {{ end }} {{- if eq ($.Scratch.Get "isComments") false -}} {{- $.Scratch.Set "isDisqus" false -}} {{- $.Scratch.Set "isGiscus" false -}} {{ end }} {{- if eq ($.Scratch.Get "isDisqus") true -}} {{- partial "disqus.html" . -}} {{- else if eq ($.Scratch.Get "isGiscus") true -}} {{- partial "giscus.html" . -}} {{- end -}}
